二分查找

2015-05-02

二分查找是最简单的查找算法，有递归和非递归的实现方式

递归

public static int recursionSearch(int [] array, int low, int high, int target){
    if(array == null || array.length < 1 || low > high)
        return -1;
    int mid = (low + high) >> 1;
    if(array[mid] == target)
        return mid;
    else if(array[mid] > target)
        return recursionSearch(array, low, mid - 1, target);
    else
        return recursionSearch(array, mid + 1, high, target);
}

非递归

public static int nonRecursionSearch(int [] array, int target){
    if(array == null || array.length < 1)
        return -1;
    int low = 0;
    int high = array.length - 1;
    while(low <= high){
        int mid = (low + high) >> 1;
        if(array[mid] == target)
            return mid;
        else if(array[mid] > target)
            high = mid - 1;
        else
            low = mid + 1;
    }
    return -1;
}
